contract SimpleStorage {
    uint storedData;

    function set(uint x) public {
        storedData = x;
    }

    function get() public view returns (uint) {
        return storedData;
    }
}

### 最后结论：当智能合约中参数类型是uint的时候，在node后台 转换的时候是 uint256,而不是该死的uint
